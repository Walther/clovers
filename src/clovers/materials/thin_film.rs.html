3use core::f32::consts::PI;
4
5use crate::{wavelength::Wavelength, Direction, Float, HitRecord};
6
7#[derive(Clone, Debug)]
8/// An iridescence feature based on thin-film interference.
9#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
10#[cfg_attr(feature = "serde-derive", serde(default))]
11pub struct ThinFilm {
12    /// Refractive index of the material.
13    pub refractive_index: Float,
14    /// Thickness of the film in nanometers.
15    pub thickness: Float,
16}
17
18impl ThinFilm {
19    /// Creates a new instance of [`ThinFilm`] with the specified `refractive_index` and `thickness` in nanometers.
20    #[must_use]
21    pub fn new(refractive_index: Float, thickness: Float) -> Self {
22        Self {
23            refractive_index,
24            thickness,
25        }
26    }
27
28    /// Calculates the strength of the interference. This should be used as a multiplier to the material's albedo. Range: `0..2` inclusive, with area 1, preserving energy conversation across spectrum.
29    #[must_use]
30    #[allow(clippy::cast_precision_loss)]
31    pub fn interference(
32        &self,
33        direction: Direction,
34        wavelength: Wavelength,
35        hit_record: &HitRecord,
36    ) -> Float {
37        // Assume ray coming from air
38        // TODO: other material interfaces?
39        let n1 = 1.0;
40        let n2 = self.refractive_index;
41        let n_ratio = n1 / n2;
42
43        // https://en.wikipedia.org/wiki/Snell%27s_law#Vector_form
44        let cos_theta_1: Float = -direction.dot(&hit_record.normal);
45        let sin_theta_1: Float = (1.0 - cos_theta_1 * cos_theta_1).sqrt();
46        let sin_theta_2: Float = n_ratio * sin_theta_1;
47        let cos_theta_2: Float = (1.0 - (sin_theta_2 * sin_theta_2)).sqrt();
48
49        // https://en.wikipedia.org/wiki/Thin-film_interference
50        let optical_path_difference = 2.0 * self.refractive_index * self.thickness * cos_theta_2;
51        let m = optical_path_difference / (wavelength as Float);
52        // range 0 to 2, area 1
53        1.0 + (m * 2.0 * PI).cos()
54    }
55}
56
57impl Default for ThinFilm {
58    fn default() -> Self {
59        Self {
60            thickness: 500.0,
61            refractive_index: 1.5,
62        }
63    }
