27#[derive(Clone, Debug)]
28#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
29/// A dispersive glass material.
30pub struct Dispersive {
31    /// Cauchy coefficient A of the material
32    #[cfg_attr(feature = "serde-derive", serde(default = "default_a"))]
33    pub cauchy_a: Float,
34    /// Cauchy coefficient B of the material
35    #[cfg_attr(feature = "serde-derive", serde(default = "default_b"))]
36    pub cauchy_b: Float,
37}
38
39fn default_a() -> Float {
40    1.5046
41}
42
43fn default_b() -> Float {
44    0.00420
45}
46
47// TODO: less precision loss?
48#[allow(clippy::cast_precision_loss)]
49impl Dispersive {
50    /// Creates a new [Dispersive] material with the given Cauchy equation constants.
51    #[must_use]
52    pub fn new(cauchy_a: Float, cauchy_b: Float) -> Self {
53        Dispersive { cauchy_a, cauchy_b }
54    }
55
56    /// Calculates the refractive index of the material for the given wavelength
57    #[must_use]
58    pub fn refractive_index(&self, wavelength: Wavelength) -> Float {
59        let wave_micros = wavelength as Float / 1000.0;
60        self.cauchy_a + (self.cauchy_b / (wave_micros * wave_micros))
61    }
62}
63
64impl Default for Dispersive {
65    fn default() -> Self {
66        Dispersive::new(default_a(), default_b())
67    }
68}

70impl MaterialTrait for Dispersive {
71    fn scatter(
72        &self,
73        ray: &Ray,
74        hit_record: &HitRecord,
75        rng: &mut SmallRng,
76    ) -> Option<ScatterRecord<'_>> {
77        // Calculate refractive index based on the wavelength of the incoming material
78        let refractive_index = self.refractive_index(ray.wavelength);
79        let refraction_ratio: Float = if hit_record.front_face {
80            1.0 / refractive_index
81        } else {
82            refractive_index
83        };
84
85        // Copied from Dielectric, is this correct?
86        let direction: Direction = ray.direction;
87        let cos_theta: Float = (-direction.dot(&hit_record.normal)).min(1.0);
88        let sin_theta: Float = (1.0 - cos_theta * cos_theta).sqrt();
89        let specular_direction: Direction = if refraction_ratio * sin_theta > 1.0 {
90            reflect(direction, hit_record.normal)
91        } else {
92            let reflect_probability: Float = schlick(cos_theta, refraction_ratio);
93            if rng.random::<Float>() < reflect_probability {
94                reflect(direction, hit_record.normal)
95            } else {
96                // Refracted
97                refract(direction, hit_record.normal, refraction_ratio)
98            }
99        };
100        let specular_ray = Ray {
101            origin: hit_record.position,
102            direction: specular_direction,
103            time: ray.time,
104            wavelength: ray.wavelength,
105        };
106
107        Some(ScatterRecord {
108            material_type: MaterialType::Specular,
109            specular_ray: Some(specular_ray),
110            pdf_ptr: PDF::ZeroPDF(ZeroPDF::new()), //TODO: ugly hack due to nullptr in original tutorial
111        })
112        // End copied
113    }
114
115    // TODO: should this material provide a `scattering_pdf` function?
116
117    fn is_wavelength_dependent(&self) -> bool {
118        true
119    }
120
121    fn color(&self, _ray: &Ray, _wavelength: Wavelength, _hit_record: &HitRecord) -> Float {
122        1.0
123    }
