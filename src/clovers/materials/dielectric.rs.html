15#[derive(Clone, Debug)]
16#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
17/// A dielectric material. This resembles glass and other transparent and reflective materials.
18pub struct Dielectric {
19    /// Refractive index of the material. Used for calculating the new direction of a ray when entering the material at an angle. Follows Snell's law of refraction. Default value: 1.5, based on typical window glass.
20    #[cfg_attr(feature = "serde-derive", serde(default = "default_index"))]
21    pub refractive_index: Float,
22    /// Color of the material. Used for colorizing the rays. Default value: [`(1.0, 1.0, 1.0)`], producing a fully transparent, clear glass.
23    #[cfg_attr(feature = "serde-derive", serde(default = "default_color"))]
24    pub color: Xyz<E>,
25}
26
27fn default_index() -> Float {
28    1.5
29}
30
31fn default_color() -> Xyz<E> {
32    Xyz::new(1.0, 1.0, 1.0)
33}
34
35impl MaterialTrait for Dielectric {
36    /// Scatter method for the Dielectric material. Given a `ray` and a `hit_record`, evaluate a [`ScatterRecord`] based on possible reflection or refraction.
37    fn scatter(
38        &self,
39        ray: &Ray,
40        hit_record: &HitRecord,
41        rng: &mut SmallRng,
42    ) -> Option<ScatterRecord<'_>> {
43        let refraction_ratio: Float = if hit_record.front_face {
44            1.0 / self.refractive_index
45        } else {
46            self.refractive_index
47        };
48
49        let direction: Direction = ray.direction;
50        let cos_theta: Float = (-direction.dot(&hit_record.normal)).min(1.0);
51        let sin_theta: Float = (1.0 - cos_theta * cos_theta).sqrt();
52        let specular_direction: Direction = if refraction_ratio * sin_theta > 1.0 {
53            reflect(direction, hit_record.normal)
54        } else {
55            let reflect_probability: Float = schlick(cos_theta, refraction_ratio);
56            if rng.random::<Float>() < reflect_probability {
57                reflect(direction, hit_record.normal)
58            } else {
59                // Refracted
60                refract(direction, hit_record.normal, refraction_ratio)
61            }
62        };
63        let specular_ray = Ray {
64            origin: hit_record.position,
65            direction: specular_direction,
66            time: ray.time,
67            wavelength: ray.wavelength,
68        };
69
70        Some(ScatterRecord {
71            material_type: MaterialType::Specular,
72            specular_ray: Some(specular_ray),
73            pdf_ptr: PDF::ZeroPDF(ZeroPDF::new()), //TODO: ugly hack due to nullptr in original tutorial
74        })
75    }
76
77    // TODO: should this material provide a `scattering_pdf` function?
78
79    fn color(&self, _ray: &Ray, wavelength: Wavelength, _hit_record: &HitRecord) -> Float {
80        spectral_power(self.color, wavelength)
81    }
82}
