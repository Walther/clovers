13/// A cone light material. The material emits light if the incoming ray is within a certain amount of degrees from the surface normal.
14#[derive(Clone, Debug)]
15#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
16pub struct ConeLight {
17    spread: Float,
18    emit: Texture,
19}
20
21impl Default for ConeLight {
22    /// Creates a new [`ConeLight`] with white light at intensity `100.0` and a spread of 10 degrees.
23    fn default() -> Self {
24        ConeLight {
25            spread: 10.0,
26            emit: Texture::SolidColor(SolidColor::new(Xyz::new(100.0, 100.0, 100.0))),
27        }
28    }
29}
30
31impl MaterialTrait for ConeLight {
32    /// Scatter method for the [`ConeLight`] material. Always returns `None`, as diffuse light does not scatter.
33    fn scatter(
34        &self,
35        _ray: &Ray,
36        _hit_record: &HitRecord,
37        _rng: &mut SmallRng,
38    ) -> Option<ScatterRecord<'_>> {
39        None
40    }
41
42    /// Emission function for [`ConeLight`]. If the given [`HitRecord`] has been hit on the `front_face`, emit a color based on the texture and surface coordinates. Otherwise, emit pure black.
43    fn emit(&self, ray: &Ray, wavelength: Wavelength, hit_record: &HitRecord) -> Float {
44        // If we don't hit the front face, return black
45        if !hit_record.front_face {
46            return 0.0;
47        }
48
49        // We have hit the front. Calculate the angle of incidence
50        let spread_radians = self.spread.to_radians();
51        let angle = (-ray.direction.dot(&hit_record.normal)
52            / (ray.direction.magnitude() * hit_record.normal.magnitude()))
53        .acos();
54
55        if angle > spread_radians {
56            return 0.0;
57        }
58
59        // HACK: current scene file format allows non-illuminants as light sources. These textures have no `emit()`, use `color()` instead.
60        match self.emit {
61            Texture::SolidColor(_) | Texture::SpatialChecker(_) | Texture::SurfaceChecker(_) => {
62                self.emit.color(ray, wavelength, hit_record)
63            }
64            _ => self.emit.emit(ray, wavelength, hit_record),
65        }
66    }
67
68    fn color(&self, ray: &Ray, wavelength: Wavelength, hit_record: &HitRecord) -> Float {
69        self.emit.color(ray, wavelength, hit_record).clamp(0.0, 1.0)
70    }
71}
72
73impl ConeLight {
74    /// Creates a new [`ConeLight`] material with the given [Texture].
75    #[must_use]
76    pub fn new(spread: Float, emit: impl Into<Texture>) -> Self {
77        ConeLight {
78            spread,
79            emit: emit.into(),
80        }
81    }
