16#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
17/// A metal material. The amount of reflection can be adjusted with the `fuzz` parameter.
18pub struct Metal {
19    #[cfg_attr(feature = "serde-derive", serde(default))]
20    albedo: Texture,
21    #[cfg_attr(feature = "serde-derive", serde(default))]
22    fuzz: Float,
23}
24
25impl MaterialTrait for Metal {
26    /// Scatter function for the [Metal] material. Metal always reflects, and a specular ray is calculated with some randomness adjusted by the `fuzz` factor. This means the metal can be made more shiny or more matte. The returned [`ScatterRecord`] will have a probability density function of [`ZeroPDF`] and material type of [`MaterialType::Specular`]
27    fn scatter(
28        &self,
29        ray: &Ray,
30        hit_record: &HitRecord,
31        rng: &mut SmallRng,
32    ) -> Option<ScatterRecord<'_>> {
33        let reflected: Direction = reflect(ray.direction, hit_record.normal);
34        let direction = *reflected + self.fuzz * *random_unit_vector(rng);
35        let direction = Unit::new_normalize(direction);
36        Some(ScatterRecord {
37            specular_ray: Some(Ray {
38                origin: hit_record.position,
39                direction,
40                time: ray.time,
41                wavelength: ray.wavelength,
42            }),
43            material_type: MaterialType::Specular,
44            pdf_ptr: PDF::ZeroPDF(ZeroPDF::new()),
45        })
46    }
47
48    // TODO: should this material provide a `scattering_pdf` function?
49    fn color(&self, ray: &Ray, wavelength: Wavelength, hit_record: &HitRecord) -> Float {
50        self.albedo.color(ray, wavelength, hit_record)
51    }
52}
53
54impl Metal {
55    /// Creates a new [Metal] material with the albedo of the given [Texture] and a smoothness-roughness factor specified by `fuzz` parameter.
56    #[must_use]
57    pub fn new(albedo: impl Into<Texture>, fuzz: Float) -> Self {
58        Metal {
59            albedo: albedo.into(),
60            fuzz: fuzz.min(1.0),
61        }
62    }
