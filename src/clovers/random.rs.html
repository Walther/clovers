1//! Various internal helper functions for getting specific kinds of random values.
2
3use crate::{Direction, Float, Vec2, Vec3, PI};
4use nalgebra::Unit;
5use rand::rngs::SmallRng;
6use rand::Rng;
7use rand_distr::{Distribution, UnitDisc, UnitSphere};
8
9/// Internal helper.
10#[must_use]
11pub fn random_unit_vector(rng: &mut SmallRng) -> Direction {
12    let v = UnitSphere.sample(rng).into();
13    Unit::new_normalize(v)
14}
15
16/// Internal helper.
17#[must_use]
18pub fn random_in_unit_disk(rng: &mut SmallRng) -> Vec2 {
19    let v: [Float; 2] = UnitDisc.sample(rng);
20    Vec2::new(v[0], v[1])
21}
22
23/// Internal helper.
24#[must_use]
25pub fn random_cosine_direction(rng: &mut SmallRng) -> Direction {
26    let r1: Float = rng.random();
27    let r2: Float = rng.random();
28    let z = (1.0 - r2).sqrt();
29
30    let phi = 2.0 * PI * r1;
31    let x = phi.cos() * r2.sqrt();
32    let y = phi.sin() * r2.sqrt();
33
34    let v: Vec3 = Vec3::new(x, y, z);
35    Unit::new_normalize(v)
36}
37
38/// Internal helper.
39#[must_use]
40pub fn random_on_hemisphere(normal: Vec3, rng: &mut SmallRng) -> Direction {
41    let in_unit_sphere: Direction = random_unit_vector(rng);
42    if in_unit_sphere.dot(&normal) > 0.0 {
43        // In the same hemisphere as the normal
44        in_unit_sphere
45    } else {
46        -in_unit_sphere
47    }
48}
