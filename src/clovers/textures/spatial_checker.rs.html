17/// A standard checkered texture based on spatial 3D texturing.
18#[derive(Clone, Debug)]
19#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
20pub struct SpatialCheckerInit {
21    /// Uniform color for the even-numbered checkers of the texture.
22    #[cfg_attr(feature = "serde-derive", serde(default = "default_even"))]
23    pub even: ColorInit,
24    /// Uniform color for the odd-numbered checkers of the texture.
25    #[cfg_attr(feature = "serde-derive", serde(default = "default_odd"))]
26    pub odd: ColorInit,
27    /// Controls the density of the checkered pattern. Default value is 1.0, which corresponds to filling a 1.0 unit cube in the coordinate system with one color of the pattern. Even values preferred - odd values may create a visually thicker stripe due to two stripes with same color being next to each other.
28    #[cfg_attr(feature = "serde-derive", serde(default = "default_density_spatial"))]
29    pub density: Float,
30}
31
32impl From<SpatialCheckerInit> for SpatialChecker {
33    fn from(init: SpatialCheckerInit) -> Self {
34        SpatialChecker::new(init.even, init.odd, init.density)
35    }
36}

38/// A standard checkered texture based on spatial 3D texturing.
39#[derive(Clone, Debug)]
40#[cfg_attr(feature = "serde-derive", derive(serde::Serialize, serde::Deserialize))]
41#[cfg_attr(feature = "serde-derive", serde(from = "SpatialCheckerInit"))]
42pub struct SpatialChecker {
43    /// Uniform color for the even-numbered checkers of the texture.
44    #[cfg_attr(feature = "serde-derive", serde(skip))]
45    even: SPD,
46    /// Uniform color for the odd-numbered checkers of the texture.
47    #[cfg_attr(feature = "serde-derive", serde(skip))]
48    odd: SPD,
49    /// Controls the density of the checkered pattern. Default value is 1.0, which corresponds to filling a 1.0 unit cube in the coordinate system with one color of the pattern. Even values preferred - odd values may create a visually thicker stripe due to two stripes with same color being next to each other.
50    pub density: Float,
51}

53#[cfg(feature = "serde-derive")]
54fn default_even() -> ColorInit {

65#[cfg(feature = "serde-derive")]
66fn default_density_spatial() -> Float {
67    1.0

70impl SpatialChecker {
71    /// Create a new `SpatialChecker` object with the specified colors and density.
72    #[must_use]
73    pub fn new(color1: impl Into<Xyz<E>>, color2: impl Into<Xyz<E>>, density: Float) -> Self {
74        let even = SPD::new(color1.into());
75        let odd = SPD::new(color2.into());
76        SpatialChecker { even, odd, density }
77    }
78}

80impl TextureTrait for SpatialChecker {
81    /// Evaluates the color at the given spatial position coordinate. Note that the `SpatialChecker` is spatial - surface coordinates are ignored.
82    fn color(&self, _ray: &Ray, wavelength: Wavelength, hit_record: &HitRecord) -> Float {
83        let position = hit_record.position;
84        let density = self.density * PI;
85        let sines = 1.0 // cosmetic 1 for readability of following lines :)
86            * (density * position.x).sin()
87            * (density * position.y).sin()
88            * (density * position.z).sin();
89        if sines < 0.0 {
90            self.odd.get(wavelength)
91        } else {
92            self.even.get(wavelength)
93        }
94    }
