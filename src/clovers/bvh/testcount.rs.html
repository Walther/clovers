11impl BVHNode<'_> {
12    // NOTE: this must be kept in close alignment with the implementation of BVHNode::hit()!
13    // TODO: maybe move the statistics counting to the method itself? Measure the impact?
14    /// Alternate hit method that maintains a test count for the BVH traversals.
15    pub fn testcount(
16        &self,
17        depth: &mut usize,
18        ray: &Ray,
19        distance_min: Float,
20        mut distance_max: Float,
21        rng: &mut SmallRng,
22    ) -> Option<HitRecord<'_>> {
23        *depth += 1;
24
25        // If we do not hit the bounding box of current node, early return None
26        if !self.aabb.hit(ray, distance_min, distance_max) {
27            return None;
28        }
29
30        // Check the distance to the bounding boxes
31        let (left_aabb_distance, right_aabb_distance) = match (self.left.aabb(), self.right.aabb())
32        {
33            // Early returns, if there's no bounding box
34            (None, None) => return None,
35            (Some(_l), None) => {
36                return recurse(&self.left, depth, ray, distance_min, distance_max, rng)
37            }
38            (None, Some(_r)) => {
39                return recurse(&self.right, depth, ray, distance_min, distance_max, rng)
40            }
41            // If we have bounding boxes, get the distances
42            (Some(l), Some(r)) => (l.distance(ray), r.distance(ray)),
43        };
44        let (_closest_aabb_distance, furthest_aabb_distance) =
45            match (left_aabb_distance, right_aabb_distance) {
46                // Early return: neither child AABB can be hit with the ray
47                (None, None) => return None,
48                // Early return: only one child can be hit with the ray
49                (Some(_d), None) => {
50                    return recurse(&self.left, depth, ray, distance_min, distance_max, rng)
51                }
52                (None, Some(_d)) => {
53                    return recurse(&self.right, depth, ray, distance_min, distance_max, rng)
54                }
55                // Default case: both children can be hit with the ray, check the distance
56                (Some(l), Some(r)) => (Float::min(l, r), Float::max(l, r)),
57            };
58
59        // Check the closest first
60        let (closest_bvh, furthest_bvh) = if left_aabb_distance < right_aabb_distance {
61            (&self.left, &self.right)
62        } else {
63            (&self.right, &self.left)
64        };
65        let closest_bvh_hit = recurse(closest_bvh, depth, ray, distance_min, distance_max, rng);
66
67        // Do we hit the closer AABB?
68        if let Some(ref hit_record) = closest_bvh_hit {
69            // Update distance_max with the distance of confirmed hit
70            distance_max = hit_record.distance;
71            // Is the hit closer than the closest point of the other AABB?
72            if hit_record.distance < furthest_aabb_distance {
73                return Some(hit_record.clone());
74            }
75        }
76        // Otherwise, check the other child too
77        let furthest_bvh_hit = recurse(furthest_bvh, depth, ray, distance_min, distance_max, rng);
78
79        // Did we hit neither of the child nodes, one of them, or both?
80        // Return the closest thing we hit
81        match (&closest_bvh_hit, &furthest_bvh_hit) {
82            (None, None) => None,
83            (None, Some(_)) => furthest_bvh_hit,
84            (Some(_), None) => closest_bvh_hit,
85            (Some(left), Some(right)) => {
86                if left.distance < right.distance {
87                    return closest_bvh_hit;
88                }
89                furthest_bvh_hit
90            }
91        }
92    }
93}

95fn recurse<'scene>(
96    bvhnode: &'scene Hitable, // BVHNode
97    depth: &mut usize,
98    ray: &Ray,
99    distance_min: Float,
100    distance_max: Float,
101    rng: &mut SmallRng,
102) -> Option<HitRecord<'scene>> {
103    match bvhnode {
104        Hitable::BVHNode(bvh) => bvh.testcount(depth, ray, distance_min, distance_max, rng),
105        hitable => hitable.hit(ray, distance_min, distance_max, rng),
106    }
