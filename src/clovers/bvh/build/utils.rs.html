3use core::cmp::Ordering;
4
5use crate::{aabb::AABB, hitable::Hitable, hitable::HitableTrait};

7pub(crate) fn box_compare(a: &Hitable, b: &Hitable, axis: usize) -> Ordering {
8    let box_a: Option<&AABB> = a.aabb();
9    let box_b: Option<&AABB> = b.aabb();
10
11    if let (Some(box_a), Some(box_b)) = (box_a, box_b) {
12        if box_a.axis(axis).min < box_b.axis(axis).min {
13            Ordering::Less
14        } else {
15            // Default to greater, even if equal
16            Ordering::Greater
17        }
18    } else {
19        panic!("No bounding box to compare with.")
20    }
21}
22
23fn box_x_compare(a: &Hitable, b: &Hitable) -> Ordering {
24    box_compare(a, b, 0)
25}
26
27fn box_y_compare(a: &Hitable, b: &Hitable) -> Ordering {
28    box_compare(a, b, 1)
29}
30
31fn box_z_compare(a: &Hitable, b: &Hitable) -> Ordering {
32    box_compare(a, b, 2)
33}
34
35pub(crate) fn get_comparator(axis: usize) -> fn(&Hitable, &Hitable) -> Ordering {
36    let comparators = [box_x_compare, box_y_compare, box_z_compare];
37    comparators[axis]
38}

40// TODO: inefficient, O(n) *and* gets called at every iteration of BVHNode creation => quadratic behavior
41#[must_use]
42pub(crate) fn vec_bounding_box(vec: &Vec<Hitable>) -> Option<AABB> {
43    if vec.is_empty() {
44        return None;
45    }
46
47    // Mutable AABB that we grow from zero
48    let mut output_box: Option<AABB> = None;
49
50    // Go through all the objects, and expand the AABB
51    for object in vec {
52        // Check if the object has a box
53        let Some(bounding) = object.aabb() else {
54            // No box found for the object, early return.
55            // Having even one unbounded object in a list makes the entire list unbounded!
56            return None;
57        };
58
59        // Do we have an output_box already saved?
60        match output_box {
61            // If we do, expand it & recurse
62            Some(old_box) => {
63                output_box = Some(AABB::combine(&old_box, bounding));
64            }
65            // Otherwise, set output box to be the newly-found box
66            None => {
67                output_box = Some(bounding.clone());
68            }
69        }
70    }
71
72    // Return the final combined output_box
73    output_box
74}
