4    aabb::AABB,

7    Float,
8};

12pub fn build(mut hitables: Vec<Hitable>) -> BVHNode {
13    // Initialize two child nodes
14    let left: Box<Hitable>;
15    let right: Box<Hitable>;
16
17    // What is the axis with the largest span?
18    // TODO: horribly inefficient, improve!
19    let bounding: AABB = vec_bounding_box(&hitables).expect("No bounding box for objects");
20    let spans = [bounding.x.size(), bounding.y.size(), bounding.z.size()];
21    let largest = Float::max(Float::max(spans[0], spans[1]), spans[2]);
22    #[allow(clippy::float_cmp)] // TODO: better code for picking the largest axis...
23    let axis: usize = spans.iter().position(|&x| x == largest).unwrap();
24    let comparator = get_comparator(axis);
25
26    // How many objects do we have?
27    let object_span = hitables.len();
28
29    if object_span == 1 {
30        // If we only have one object, add one and an empty object.
31        // TODO: can this hack be removed?
32        left = Box::new(hitables.remove(0));
33        right = Box::new(Hitable::Empty(Empty {}));
34        let aabb = left.aabb().unwrap().clone(); // TODO: remove unwrap
35        return BVHNode { left, right, aabb };
36    } else if object_span == 2 {
37        // If we are comparing two objects, perform the comparison
38        // Insert the child nodes in order
39        match comparator(&hitables[0], &hitables[1]) {
40            Ordering::Less => {
41                left = Box::new(hitables.remove(0));
42                right = Box::new(hitables.remove(0)); // Second item is at index 0 after first removal
43            }
44            Ordering::Greater => {
45                left = Box::new(hitables.remove(1));
46                right = Box::new(hitables.remove(0));
47            }
48            Ordering::Equal => {
49                // TODO: what should happen here?
50                panic!("Equal objects in BVHNode from_list");
51            }
52        }
53    } else if object_span == 3 {
54        // Three objects: create one bare object and one BVHNode with two objects
55        hitables.sort_by(comparator);
56        left = Box::new(hitables.remove(0));
57        let rl = hitables.remove(0);
58        let rr = hitables.remove(0);
59        let aabb = AABB::combine(rl.aabb().unwrap(), rr.aabb().unwrap());
60        right = Box::new(Hitable::BVHNode(BVHNode {
61            left: Box::new(rl),
62            right: Box::new(rr),
63            aabb,
64        }));
65    } else {
66        // Otherwise, recurse
67        hitables.sort_by(comparator);
68
69        // Split the vector; divide and conquer
70        let mid = object_span / 2;
71        let hitables_right = hitables.split_off(mid);
72        left = Box::new(Hitable::BVHNode(build(hitables)));
73        right = Box::new(Hitable::BVHNode(build(hitables_right)));
74    }
75
76    let box_left = left.aabb();
77    let box_right = right.aabb();
78
79    // Generate a bounding box and BVHNode if possible
80    if let (Some(box_left), Some(box_right)) = (box_left, box_right) {
81        let aabb = AABB::combine(box_left, box_right);
82
83        BVHNode { left, right, aabb }
84    } else {
85        panic!("No bounding box in bvh_node constructor");
86    }
87}
