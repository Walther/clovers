17/// Heavily inspired by the wonderful blog series <https://jacco.ompf2.com/2022/04/18/how-to-build-a-bvh-part-2-faster-rays/>.
18pub fn build(mut hitables: Vec<Hitable>) -> BVHNode {
19    // Initialize two child nodes
20    let left: Box<Hitable>;
21    let right: Box<Hitable>;
22
23    let aabb = vec_bounding_box(&hitables).unwrap();
24    let count = hitables.len();
25
26    // Possible leaf nodes
27    match count {
28        0 => {
29            #[cfg(feature = "tracing")]
30            warn!("building a BVHNode from zero hitables");
31            left = Box::new(Hitable::Empty(Empty {}));
32            right = Box::new(Hitable::Empty(Empty {}));
33            return BVHNode { left, right, aabb };
34        }
35        1 => {
36            left = Box::new(hitables.remove(0));
37            right = Box::new(Hitable::Empty(Empty {}));
38            return BVHNode { left, right, aabb };
39        }
40        2 => {
41            left = Box::new(hitables.remove(0));
42            right = Box::new(hitables.remove(0));
43            return BVHNode { left, right, aabb };
44        }
45        _ => (),
46    }
47
48    // If we have more than two nodes, split and recurse
49    let (axis, position) = find_best_split(&hitables);
50    let (mut hitables_left, mut hitables_right): (Vec<Hitable>, Vec<Hitable>) = hitables
51        .into_iter()
52        // NOTE: match comparison in evaluate_sah
53        .partition(|hitable| hitable.centroid()[axis] <= position);
54
55    // Avoid infinite recursion
56    if hitables_left.is_empty() {
57        #[cfg(feature = "tracing")]
58        warn!("hitables_left is empty, bvh tree might become deep");
59        let h = hitables_right.remove(0);
60        left = Box::new(h);
61        right = Box::new(Hitable::BVHNode(build(hitables_right)));
62
63        return BVHNode { left, right, aabb };
64    }
65    if hitables_right.is_empty() {
66        #[cfg(feature = "tracing")]
67        warn!("hitables_right is empty, bvh tree might become deep");
68        let h = hitables_left.remove(0);
69        left = Box::new(Hitable::BVHNode(build(hitables_left)));
70        right = Box::new(h);
71
72        return BVHNode { left, right, aabb };
73    }
74
75    left = Box::new(Hitable::BVHNode(build(hitables_left)));
76    right = Box::new(Hitable::BVHNode(build(hitables_right)));
77
78    BVHNode { left, right, aabb }
79}

81fn find_best_split(hitables: &Vec<Hitable>) -> (usize, Float) {
82    // TODO: configurable?
83    const SPLIT_COUNT: u8 = 8;
84    const SPLIT_COUNT_F: Float = SPLIT_COUNT as Float;
85
86    #[cfg(feature = "tracing")]
87    if hitables.len() == 1 {
88        warn!("best_split trying to split a single hitable");
89    }
90
91    let mut found = false;
92    let mut best_axis = 0;
93    let mut best_pos = 0.0;
94    let mut best_cost = Float::INFINITY;
95
96    for axis in 0..3 {
97        // find the splitting bounds based on the centroids of the hitables
98        // this is better than using the bounding box of the hitables
99        // because the bounding box can be much larger due to the size of the objects
100        let mut bounds_min = Float::INFINITY;
101        let mut bounds_max = Float::NEG_INFINITY;
102        for hitable in hitables {
103            bounds_min = Float::min(bounds_min, hitable.centroid()[axis]);
104            bounds_max = Float::max(bounds_max, hitable.centroid()[axis]);
105        }
106
107        #[allow(clippy::float_cmp)]
108        if bounds_min == bounds_max {
109            continue;
110        }
111
112        let scale = (bounds_max - bounds_min) / SPLIT_COUNT_F;
113        for i in 0..SPLIT_COUNT {
114            let candidate_pos = bounds_min + Float::from(i) * scale;
115            let cost = evaluate_sah(hitables, axis, candidate_pos);
116            if cost < best_cost {
117                found = true;
118                best_pos = candidate_pos;
119                best_axis = axis;
120                best_cost = cost;
121            }
122        }
123    }
124
125    // TODO: fix this, if possible!
126    #[cfg(feature = "tracing")]
127    if !found {
128        warn!("best_split did not find an improved split, returning defaults!");
129    }
130
131    (best_axis, best_pos)
132}

134fn evaluate_sah(hitables: &Vec<Hitable>, axis: usize, position: Float) -> Float {
135    // determine triangle counts and bounds for this split candidate
136    let mut left_box = AABB::default();
137    let mut right_box = AABB::default();
138    // 2 * 2^64 primitives should be enough
139    let mut left_count = 0u64;
140    let mut right_count = 0u64;
141    for hitable in hitables {
142        // NOTE: match comparison in do_split
143        if hitable.centroid()[axis] <= position {
144            left_count += 1;
145            left_box = AABB::combine(&left_box, hitable.aabb().unwrap()); // TODO: remove unwrap
146        } else {
147            right_count += 1;
148            right_box = AABB::combine(&right_box, hitable.aabb().unwrap()); // TODO: remove unwrap
149        }
150    }
151    #[allow(clippy::cast_precision_loss)]
152    let cost: Float =
153        left_count as Float * left_box.area() + right_count as Float * right_box.area();
154
155    cost
156}
