14    /// The main `hit` function for a [`BVHNode`]. Given a [Ray], and an interval `distance_min` and `distance_max`, returns either `None` or `Some(HitRecord)` based on whether the ray intersects with the encased objects during that interval.
15    fn hit(
16        &self,
17        ray: &Ray,
18        distance_min: Float,
19        mut distance_max: Float,
20        rng: &mut SmallRng,
21    ) -> Option<HitRecord<'_>> {
22        // If we do not hit the bounding box of current node, early return None
23        if !self.aabb.hit(ray, distance_min, distance_max) {
24            return None;
25        }
26
27        // Check the distance to the bounding boxes
28        let (left_aabb_distance, right_aabb_distance) = match (self.left.aabb(), self.right.aabb())
29        {
30            // Early returns, if there's no bounding box
31            (None, None) => return None,
32            (Some(_l), None) => return self.left.hit(ray, distance_min, distance_max, rng),
33            (None, Some(_r)) => return self.right.hit(ray, distance_min, distance_max, rng),
34            // If we have bounding boxes, get the distances
35            (Some(l), Some(r)) => (l.distance(ray), r.distance(ray)),
36        };
37        let (_closest_aabb_distance, furthest_aabb_distance) =
38            match (left_aabb_distance, right_aabb_distance) {
39                // Early return: neither child AABB can be hit with the ray
40                (None, None) => return None,
41                // Early return: only one child can be hit with the ray
42                (Some(_d), None) => return self.left.hit(ray, distance_min, distance_max, rng),
43                (None, Some(_d)) => return self.right.hit(ray, distance_min, distance_max, rng),
44                // Default case: both children can be hit with the ray, check the distance
45                (Some(l), Some(r)) => (Float::min(l, r), Float::max(l, r)),
46            };
47
48        // Check the closest first
49        let (closest_bvh, furthest_bvh) = if left_aabb_distance < right_aabb_distance {
50            (&self.left, &self.right)
51        } else {
52            (&self.right, &self.left)
53        };
54        let closest_bvh_hit = closest_bvh.hit(ray, distance_min, distance_max, rng);
55
56        // Do we hit the closer AABB?
57        if let Some(ref hit_record) = closest_bvh_hit {
58            // Update distance_max with the distance of confirmed hit
59            distance_max = hit_record.distance;
60            // Is the hit closer than the closest point of the other AABB?
61            if hit_record.distance < furthest_aabb_distance {
62                return Some(hit_record.clone());
63            }
64        }
65        // Otherwise, check the other child too
66        let furthest_bvh_hit = furthest_bvh.hit(ray, distance_min, distance_max, rng);
67
68        // Did we hit neither of the child nodes, one of them, or both?
69        // Return the closest thing we hit
70        match (&closest_bvh_hit, &furthest_bvh_hit) {
71            (None, None) => None,
72            (None, Some(_)) => furthest_bvh_hit,
73            (Some(_), None) => closest_bvh_hit,
74            (Some(left), Some(right)) => {
75                if left.distance < right.distance {
76                    return closest_bvh_hit;
77                }
78                furthest_bvh_hit
79            }
80        }
81    }

88    /// Returns a probability density function value based on the children
89    fn pdf_value(
90        &self,
91        origin: Position,
92        direction: Direction,
93        wavelength: Wavelength,
94        time: Float,
95        rng: &mut SmallRng,
96    ) -> Float {
97        match (&*self.left, &*self.right) {
98            (_, Hitable::Empty(_)) => self
99                .left
100                .pdf_value(origin, direction, wavelength, time, rng),
101            (Hitable::Empty(_), _) => self
102                .right
103                .pdf_value(origin, direction, wavelength, time, rng),
104            (_, _) => Float::midpoint(
105                self.left
106                    .pdf_value(origin, direction, wavelength, time, rng),
107                self.right
108                    .pdf_value(origin, direction, wavelength, time, rng),
109            ),
110        }
111    }

122                } else {
123                    self.right.random(origin, rng)
