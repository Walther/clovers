14    /// The main `hit` function for a [`BVHNode`]. Given a [Ray], and an interval `distance_min` and `distance_max`, returns either `None` or `Some(HitRecord)` based on whether the ray intersects with the encased objects during that interval.
15    #[must_use]
16    fn hit(
17        &self,
18        ray: &Ray,
19        distance_min: Float,
20        mut distance_max: Float,
21        rng: &mut SmallRng,
22    ) -> Option<HitRecord> {
23        // If we do not hit the bounding box of current node, early return None
24        if !self.aabb.hit(ray, distance_min, distance_max) {
25            return None;
26        }
27
28        // Check the distance to the bounding boxes
29        let (left_aabb_distance, right_aabb_distance) = match (self.left.aabb(), self.right.aabb())
30        {
31            // Early returns, if there's no bounding box
32            (None, None) => return None,
33            (Some(_l), None) => return self.left.hit(ray, distance_min, distance_max, rng),
34            (None, Some(_r)) => return self.right.hit(ray, distance_min, distance_max, rng),
35            // If we have bounding boxes, get the distances
36            (Some(l), Some(r)) => (l.distance(ray), r.distance(ray)),
37        };
38        let (_closest_aabb_distance, furthest_aabb_distance) =
39            match (left_aabb_distance, right_aabb_distance) {
40                // Early return: neither child AABB can be hit with the ray
41                (None, None) => return None,
42                // Early return: only one child can be hit with the ray
43                (Some(_d), None) => return self.left.hit(ray, distance_min, distance_max, rng),
44                (None, Some(_d)) => return self.right.hit(ray, distance_min, distance_max, rng),
45                // Default case: both children can be hit with the ray, check the distance
46                (Some(l), Some(r)) => (Float::min(l, r), Float::max(l, r)),
47            };
48
49        // Check the closest first
50        let (closest_bvh, furthest_bvh) = if left_aabb_distance < right_aabb_distance {
51            (&self.left, &self.right)
52        } else {
53            (&self.right, &self.left)
54        };
55        let closest_bvh_hit = closest_bvh.hit(ray, distance_min, distance_max, rng);
56
57        // Do we hit the closer AABB?
58        if let Some(ref hit_record) = closest_bvh_hit {
59            // Update distance_max with the distance of confirmed hit
60            distance_max = hit_record.distance;
61            // Is the hit closer than the closest point of the other AABB?
62            if hit_record.distance < furthest_aabb_distance {
63                return Some(hit_record.clone());
64            }
65        }
66        // Otherwise, check the other child too
67        let furthest_bvh_hit = furthest_bvh.hit(ray, distance_min, distance_max, rng);
68
69        // Did we hit neither of the child nodes, one of them, or both?
70        // Return the closest thing we hit
71        match (&closest_bvh_hit, &furthest_bvh_hit) {
72            (None, None) => None,
73            (None, Some(_)) => furthest_bvh_hit,
74            (Some(_), None) => closest_bvh_hit,
75            (Some(left), Some(right)) => {
76                if left.distance < right.distance {
77                    return closest_bvh_hit;
78                }
79                furthest_bvh_hit
80            }
81        }
82    }

90    /// Returns a probability density function value based on the children
91    #[must_use]
92    fn pdf_value(
93        &self,
94        origin: Position,
95        direction: Direction,
96        wavelength: Wavelength,
97        time: Float,
98        rng: &mut SmallRng,
99    ) -> Float {
100        match (&*self.left, &*self.right) {
101            (_, Hitable::Empty(_)) => self
102                .left
103                .pdf_value(origin, direction, wavelength, time, rng),
104            (Hitable::Empty(_), _) => self
105                .right
106                .pdf_value(origin, direction, wavelength, time, rng),
107            (_, _) => {
108                (self
109                    .left
110                    .pdf_value(origin, direction, wavelength, time, rng)
111                    + self
112                        .right
113                        .pdf_value(origin, direction, wavelength, time, rng))
114                    / 2.0
115            }
116        }
117    }
