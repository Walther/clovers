33/// A moving sphere object. This is represented by one `radius`, two center points `center_0` `center_1`, two times `time_0` `time_1`, and a [Material]. Any [Rays](Ray) hitting the object will also have an internal `time` value, which will be used for determining the interpolated position of the sphere at that time. With lots of rays hitting every pixel but at randomized times, we get temporal multiplexing and an approximation of perceived motion blur.
34pub struct MovingSphere<'scene> {
35    /// Center point of the sphere at `time_0`
36    pub center_0: Position,
37    /// Center point of the sphere at `time_1`
38    pub center_1: Position,
39    /// Time 0
40    pub time_0: Float,
41    /// Time 1
42    pub time_1: Float,
43    /// Radius of the sphere
44    pub radius: Float,
45    /// Material of the sphere
46    pub material: &'scene Material,
47    /// Axis-aligned bounding box
48    pub aabb: AABB,
49}

51impl<'scene> MovingSphere<'scene> {
52    /// Creates a new `MovingSphere` object. See the struct documentation for more information: [`MovingSphere`].
53    #[must_use]
54    pub fn new(
55        center_0: Position,
56        center_1: Position,
57        time_0: Float,
58        time_1: Float,
59        radius: Float,
60        material: &'scene Material,
61    ) -> Self {
62        let box0: AABB = AABB::new_from_coords(
63            center_0 - Position::new(radius, radius, radius),
64            center_0 + Position::new(radius, radius, radius),
65        );
66        let box1: AABB = AABB::new_from_coords(
67            center_1 - Position::new(radius, radius, radius),
68            center_1 + Position::new(radius, radius, radius),
69        );
70
71        let aabb = AABB::combine(&box0, &box1);
72
73        MovingSphere {
74            center_0,
75            center_1,
76            time_0,
77            time_1,
78            radius,
79            material,
80            aabb,
81        }
82    }
83
84    /// Returns the interpolated center of the moving sphere at the given time.
85    #[must_use]
86    pub fn center(&self, time: Float) -> Position {
87        self.center_0
88            + ((time - self.time_0) / (self.time_1 - self.time_0)) * (self.center_1 - self.center_0)
89    }
90
91    /// Returns the U,V surface coordinates of a hitpoint
92    // TODO: verify this is up to date with the sphere get_uv methods and matches a surface coordinate instead of volumetric space cordinate
93    #[must_use]
94    pub fn get_uv(&self, hit_position: Position, time: Float) -> (Float, Float) {
95        let translated: Position = (hit_position - self.center(time)) / self.radius;
96        let phi: Float = translated.z.atan2(translated.x);
97        let theta: Float = translated.y.asin();
98        let u: Float = 1.0 - (phi + PI) / (2.0 * PI);
99        let v: Float = (theta + PI / 2.0) / PI;
100        (u, v)
101    }
102}

104impl HitableTrait for MovingSphere<'_> {
105    /// Hit method for the [`MovingSphere`] object. First gets the interpolated center position at the given time, then follows the implementation of [Sphere](crate::objects::Sphere) object's hit method.
106    fn hit(
107        &self,
108        ray: &Ray,
109        distance_min: Float,
110        distance_max: Float,
111        _rng: &mut SmallRng,
112    ) -> Option<HitRecord<'_>> {
113        let oc = ray.origin - self.center(ray.time);
114        let a: Float = ray.direction.norm_squared();
115        let half_b: Float = oc.dot(&ray.direction);
116        let c: Float = oc.norm_squared() - self.radius * self.radius;
117        let discriminant = half_b * half_b - a * c;
118        if discriminant > 0.0 {
119            let root: Float = discriminant.sqrt();
120            let distance: Float = (-half_b - root) / a;
121            // First possible root
122            if distance < distance_max && distance > distance_min {
123                let position: Position = ray.evaluate(distance);
124                let outward_normal = (position - self.center(ray.time)) / self.radius;
125                let outward_normal = Unit::new_normalize(outward_normal);
126                let (u, v) = self.get_uv(position, ray.time);
127                let mut record = HitRecord {
128                    distance,
129                    position,
130                    normal: outward_normal,
131                    u,
132                    v,
133                    material: self.material,
134                    front_face: false, // TODO: fix having to declare it before calling face_normal
135                };
136                record.set_face_normal(ray, outward_normal);
137                return Some(record);
138            }
139            // Second possible root
140            let distance: Float = (-half_b + root) / a;
141            if distance < distance_max && distance > distance_min {
142                let position: Position = ray.evaluate(distance);
143                let outward_normal = (position - self.center(ray.time)) / self.radius;
144                let outward_normal = Unit::new_normalize(outward_normal);
145                let (u, v) = self.get_uv(position, ray.time);
146                let mut record = HitRecord {
147                    distance,
148                    position,
149                    normal: outward_normal,
150                    u,
151                    v,
152                    material: self.material,
153                    front_face: false, // TODO: fix having to declare it before calling face_normal
154                };
155                record.set_face_normal(ray, outward_normal);
156                return Some(record);
157            }
158        }
159        None
160    }
