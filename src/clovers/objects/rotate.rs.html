37impl<'scene> RotateY<'scene> {
38    /// Creates a new `RotateY` object. It wraps the given [Object] and has adjusted `hit()` and `bounding_box()` methods based on the `angle` given.
39    #[must_use]
40    pub fn new(object: Box<Hitable<'scene>>, angle: Float) -> Self {
41        // TODO: add proper time support
42        let radians: Float = angle.to_radians();
43        let sin_theta: Float = radians.sin();
44        let cos_theta: Float = radians.cos();
45        let bounding_box: Option<&AABB> = object.aabb();
46
47        // Does our object have a bounding box?
48        let Some(bbox) = bounding_box else {
49            return RotateY {
50                object,
51                sin_theta,
52                cos_theta,
53                aabb: None,
54            };
55        };
56
57        // Start with infinite bounds
58        let mut min: Vec3 = Vec3::new(Float::INFINITY, Float::INFINITY, Float::INFINITY);
59        let mut max: Vec3 = Vec3::new(
60            Float::NEG_INFINITY,
61            Float::NEG_INFINITY,
62            Float::NEG_INFINITY,
63        );
64
65        // Calculate new bounds
66        for i in [0.0, 1.0] {
67            for j in [0.0, 1.0] {
68                for k in [0.0, 1.0] {
69                    let i_f: Float = i;
70                    let j_f: Float = j;
71                    let k_f: Float = k;
72
73                    let x: Float = i_f * bbox.x.max + (1.0 - i_f) * bbox.x.min;
74                    let y: Float = j_f * bbox.y.max + (1.0 - j_f) * bbox.y.min;
75                    let z: Float = k_f * bbox.z.max + (1.0 - k_f) * bbox.z.min;
76
77                    let new_x: Float = cos_theta * x + sin_theta * z;
78                    let new_z: Float = -sin_theta * x + cos_theta * z;
79
80                    let tester: Vec3 = Vec3::new(new_x, y, new_z);
81
82                    for c in 0..3 {
83                        min[c] = min[c].min(tester[c]);
84                        max[c] = max[c].max(tester[c]);
85                    }
86                }
87            }
88        }
89
90        // Return a Rotate object with the new bounding box and pre-calculated rotation utilities
91        RotateY {
92            object,
93            sin_theta,
94            cos_theta,
95            aabb: Some(AABB::new_from_coords(min, max)),
96        }
97    }
98}

100impl HitableTrait for RotateY<'_> {
101    /// Hit method for the [`RotateY`] object. Finds the rotation-adjusted [`HitRecord`] for the possible intersection of the [Ray] with the encased [Object].
102    fn hit(
103        &self,
104        ray: &Ray,
105        distance_min: Float,
106        distance_max: Float,
107        rng: &mut SmallRng,
108    ) -> Option<HitRecord<'_>> {
109        let mut origin: Position = ray.origin;
110        let mut direction: Vec3 = *ray.direction;
111
112        origin[0] = self.cos_theta * ray.origin[0] - self.sin_theta * ray.origin[2];
113        origin[2] = self.sin_theta * ray.origin[0] + self.cos_theta * ray.origin[2];
114
115        direction[0] = self.cos_theta * ray.direction[0] - self.sin_theta * ray.direction[2];
116        direction[2] = self.sin_theta * ray.direction[0] + self.cos_theta * ray.direction[2];
117
118        let direction = Unit::new_normalize(direction);
119
120        let rotated_r: Ray = Ray {
121            origin,
122            direction,
123            time: ray.time,
124            wavelength: ray.wavelength,
125        };
126
127        let Some(hit_record) = self.object.hit(&rotated_r, distance_min, distance_max, rng) else {
128            // Did not hit rotated object, early return None
129            return None;
130        };
131
132        // Determine where the intersection is
133        // TODO: understand and explain
134        let mut position: Position = hit_record.position;
135        let mut normal: Vec3 = *hit_record.normal;
136        let distance: Float = hit_record.distance;
137
138        position[0] =
139            self.cos_theta * hit_record.position[0] + self.sin_theta * hit_record.position[2];
140        position[2] =
141            -self.sin_theta * hit_record.position[0] + self.cos_theta * hit_record.position[2];
142
143        normal[0] = self.cos_theta * hit_record.normal[0] + self.sin_theta * hit_record.normal[2];
144        normal[2] = -self.sin_theta * hit_record.normal[0] + self.cos_theta * hit_record.normal[2];
145
146        let normal = Unit::new_normalize(normal);
147
148        let mut record = HitRecord {
149            distance,
150            position,
151            normal,
152            u: hit_record.u,
153            v: hit_record.v,
154            material: hit_record.material,
155            front_face: false, // TODO: fix having to declare it before calling face_normal
156        };
157        record.set_face_normal(&rotated_r, normal);
158        Some(record)
159    }
