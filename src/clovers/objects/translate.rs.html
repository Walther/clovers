28/// Translate object. It wraps the given [Object] and has adjusted `hit()` and `bounding_box()` methods based on the `offset` given.

35impl<'scene> Translate<'scene> {
36    /// Creates a new `Translate` object. It wraps the given [Object] and has adjusted `hit()` and `bounding_box()` methods based on the `offset` given.
37    ///
38    /// # Panics
39    /// This method may panic if the given object does not have a valid `AABB`.
40    #[must_use]
41    pub fn new(object: Box<Hitable<'scene>>, offset: Vec3) -> Self {
42        // TODO: time
43        let aabb = object.aabb().unwrap().clone() + offset;
44        Translate {
45            object,
46            offset,
47            aabb,
48        }
49    }
50}

52impl HitableTrait for Translate<'_> {
53    /// Hit method for the [Translate] object. Finds the translation-adjusted [`HitRecord`] for the possible intersection of the [Ray] with the encased [Object].
54    fn hit(
55        &self,
56        ray: &Ray,
57        distance_min: Float,
58        distance_max: Float,
59        rng: &mut SmallRng,
60    ) -> Option<HitRecord<'_>> {
61        let moved_ray = Ray {
62            origin: ray.origin - self.offset,
63            direction: ray.direction,
64            time: ray.time,
65            wavelength: ray.wavelength,
66        };
67
68        match self.object.hit(&moved_ray, distance_min, distance_max, rng) {
69            // Didn't hit anything, return None
70            None => None,
71            // Hit something, adjust the position and normal
72            Some(mut hit_record) => {
73                hit_record.position += self.offset;
74                hit_record.set_face_normal(&moved_ray, hit_record.normal);
75                Some(hit_record)
76            }
77        }
78    }
79
80    /// Bounding box method for the [Translate] object. Finds the axis-aligned bounding box [AABB] for the encased [Object] after adjusting for translation.
81    fn aabb(&self) -> Option<&AABB> {
82        Some(&self.aabb)
83    }
84
85    /// Returns a probability density function value based on the inner object
86    fn pdf_value(
87        &self,
88        origin: Position,
89        direction: Direction,
90        wavelength: Wavelength,
91        time: Float,
92        rng: &mut SmallRng,
93    ) -> Float {
94        // TODO: is this correct?
95        self.object
96            .pdf_value(origin + self.offset, direction, wavelength, time, rng)
97    }
98
99    // TODO: correctness
100    fn centroid(&self) -> Position {
101        self.object.centroid() + self.offset
102    }
