1//! The very core of the ray tracing rendering itself: the [Ray]
2
3use crate::{wavelength::Wavelength, Direction, Float, Position};
4
5/// A Ray has an origin and a direction, as well as an instant in time it exists in. Motion blur is achieved by creating multiple rays with slightly different times.
6#[derive(Clone, Debug, PartialEq)]
7pub struct Ray {
8    /// The origin of the ray.
9    pub origin: Position,
10    /// The direction of the ray.
11    pub direction: Direction,
12    /// The time instant at which the ray exists.
13    pub time: Float,
14    /// Wavelength of the ray
15    pub wavelength: Wavelength,
16}
17
18impl Ray {
19    /// Evaluates the position (coordinate) at which the ray is at the given parameter, considering the origin and direction. Considering a default unit speed of 1 per unit time, this function can be given either a time or a distance.
20    #[must_use]
21    pub fn evaluate(&self, parameter: Float) -> Position {
22        self.origin + parameter * *self.direction
23    }
24}
