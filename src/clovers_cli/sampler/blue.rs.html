1//! A sampler based on blue noise. Works especially well at low samples-per-pixel counts.
2//!
3//! Utilizes library code from <https://github.com/Jasper-Bekkers/blue-noise-sampler>.
4
5use clovers::{wavelength::sample_wavelength, Float, Vec2, PI};
6
7use super::{Randomness, SamplerDimension, SamplerTrait};
8
9pub struct BlueSampler {
10    get: fn(i32, i32, i32, SamplerDimension) -> Float,
11}
12
13impl BlueSampler {
14    pub fn new(samples: u32) -> Self {
15        let get = match samples {
16            1 => blue_sample_spp1,
17            2 => blue_sample_spp2,
18            4 => blue_sample_spp4,
19            8 => blue_sample_spp8,
20            16 => blue_sample_spp16,
21            32 => blue_sample_spp32,
22            64 => blue_sample_spp64,
23            128 => blue_sample_spp128,
24            256 => blue_sample_spp256,
25            _ => unimplemented!(
26                "blue sampler only supports sample-per-pixel counts that are powers of two of and up to 256"
27            ),
28        };
29        Self { get }
30    }
31}
32
33impl SamplerTrait<'_> for BlueSampler {
34    fn sample(&mut self, i: i32, j: i32, index: i32) -> Randomness {
35        let pixel_offset = Vec2::new(
36            (self.get)(i, j, index, SamplerDimension::PixelOffsetX),
37            (self.get)(i, j, index, SamplerDimension::PixelOffsetY),
38        );
39        let lens_offset = in_unit_disk(
40            (self.get)(i, j, index, SamplerDimension::LensOffsetX),
41            (self.get)(i, j, index, SamplerDimension::LensOffsetY),
42        );
43        let time = (self.get)(i, j, index, SamplerDimension::Time);
44        // TODO: verify uniformity & correctness for math?
45        let wavelength = sample_wavelength((self.get)(i, j, index, SamplerDimension::Wavelength));
46
47        Randomness {
48            pixel_offset,
49            lens_offset,
50            time,
51            wavelength,
52        }
53    }
54
55    fn sample_dimension(
56        &mut self,
57        i: i32,
58        j: i32,
59        index: i32,
60        dimension: SamplerDimension,
61    ) -> Float {
62        (self.get)(i, j, index, dimension)
63    }
64}

66macro_rules! define_blue_sampler {
67    ($spp:ident) => {
68        ::paste::paste! {
69            pub fn [<blue_sample_ $spp>](
70                mut pixel_i: i32,
71                mut pixel_j: i32,
72                mut sample_index: i32,
73                sample_dimension: SamplerDimension) -> Float {
74                    let mut sample_dimension = sample_dimension as i32;
75
76                    use blue_noise_sampler::$spp::*;
77
78                    // Adapted from <https://dl.acm.org/doi/10.1145/3306307.3328191> and <https://github.com/Jasper-Bekkers/blue-noise-sampler>
79
80                    // wrap arguments
81                    pixel_i &= 127;
82                    pixel_j &= 127;
83                    sample_index &= 255;
84                    sample_dimension &= 255;
85
86                    // xor index based on optimized ranking
87                    // jb: 1spp blue noise has all 0 in g_blueNoiseRankingTile so we can skip the load
88                    let index = sample_dimension + (pixel_i + pixel_j * 128) * 8;
89                    let index = index as usize;
90                    let ranked_sample_index = sample_index ^ RANKING_TILE[index];
91
92                    // fetch value in sequence
93                    let index = sample_dimension + ranked_sample_index * 256;
94                    let index = index as usize;
95                    let value = SOBOL[index];
96
97                    // If the dimension is optimized, xor sequence value based on optimized scrambling
98                    let index = (sample_dimension % 8) + (pixel_i + pixel_j * 128) * 8;
99                    let index = index as usize;
100                    let value = value ^ SCRAMBLING_TILE[index];
101
102                    // convert to float and return
103                    let v: Float = (0.5 + value as Float) / 256.0;
104                    v
105            }
106        }
107    };
108}

120/// Given two samples in range `[0..1]`, return a sample within `[-0.5 .. 0.5]` unit disk.
121/// Based on <https://stackoverflow.com/a/50746409>
122fn in_unit_disk(x: Float, y: Float) -> Vec2 {
123    // Polar coordinates + correcting for the distribution using sqrt
124    let r = x.sqrt();
125    let theta = y * 2.0 * PI;
126    // Conversion to Cartesian coordinates
127    let x = r * theta.cos();
128    let y = r * theta.sin();
129    Vec2::new(x, y)
130}
