1//! Alternative rendering method. Only returns a normalmap of the image, colorized using standard conventions.
2
3use clovers::{
4    hitable::HitableTrait, ray::Ray, scenes::Scene, Direction, Float, Vec3, EPSILON_SHADOW_ACNE,
5};
6use palette::LinSrgb;
7use rand::rngs::SmallRng;
8
9/// Rendering function for getting a normal map in tangent space. Sends a [Ray] to the [Scene], sees what it hits, gets the normal at that point, and returns a color based on the normal mapping colorization. Wikipedia: [Normal mapping](https://en.wikipedia.org/wiki/Normal_mapping).
10#[must_use]
11pub fn normal_map(ray: &Ray, scene: &Scene, rng: &mut SmallRng) -> LinSrgb {
12    let Some(hit_record) = scene
13        .bvh_root
14        .hit(ray, EPSILON_SHADOW_ACNE, Float::MAX, rng)
15    else {
16        // If the ray hits nothing, early return black
17        return LinSrgb::new(0.0, 0.0, 0.0);
18    };
19
20    let normal: Direction = hit_record.normal;
21    normal_to_color(normal)
22}
23
24/// Given a surface normal, return a color based on normal mapping colorization.
25#[must_use]
26pub fn normal_to_color(normal: Direction) -> LinSrgb {
27    // flip the Z and X axes because the wikipedia example uses left-handed coordinate system and my renderer uses a right-handed one for some reason.
28    // TODO: figure out a good coordinate system to use... See also https://twitter.com/FreyaHolmer/status/1325556229410861056
29    let normal: Vec3 = Vec3::new(-normal.x, normal.y, -normal.z);
30    // TODO: verify correctness
31    let r = 0.5 + 0.5 * normal.x; // X -1 to 1 = 0.0 to 1.0
32    let g = 0.5 + 0.5 * normal.y; // Y -1 to 1 = 0.0 to 1.0
33                                  // Z  0 to 1 = 0.0 to 1.0
34    let b = if normal.z < 0.0 { 0.0 } else { normal.z };
35
36    LinSrgb::new(r, g, b)
37}
