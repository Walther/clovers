1//! Alternative rendering methods for debug visualization purposes.
2
3use clovers::{ray::Ray, scenes::Scene, Float, EPSILON_SHADOW_ACNE};
4use palette::{chromatic_adaptation::AdaptInto, white_point::E, LinSrgb, Xyz};
5use rand::rngs::SmallRng;
6
7/// Visualizes the BVH traversal count - how many BVH nodes needed to be tested for intersection?
8#[must_use]
9pub fn bvh_testcount(ray: &Ray, scene: &Scene, rng: &mut SmallRng) -> Xyz<E> {
10    let mut depth = 0;
11    scene
12        .bvh_root
13        .testcount(&mut depth, ray, EPSILON_SHADOW_ACNE, Float::MAX, rng);
14
15    bvh_testcount_to_color(depth)
16}
17
18#[must_use]
19pub fn bvh_testcount_to_color(depth: usize) -> Xyz<E> {
20    let color: LinSrgb = match depth {
21        // under 256, grayscale
22        0..=255 => {
23            let depth = depth as Float / 255.0;
24            LinSrgb::new(depth, depth, depth)
25        }
26        // more than 256, yellow
27        256..=511 => LinSrgb::new(1.0, 1.0, 0.0),
28        // more than 512, orange
29        512..=1023 => LinSrgb::new(1.0, 0.5, 0.0),
30        // more than 1024, red
31        1024.. => LinSrgb::new(1.0, 0.0, 0.0),
32    };
33
34    color.adapt_into()
35}
