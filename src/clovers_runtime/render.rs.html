16#[derive(Args, Clone, Debug)]
17pub struct RenderOptions {
18    /// Input filename / location
19    #[arg(short, long)]
20    pub input: String,
21    /// Output file path, without extension. Defaults to `./renders/unix_timestamp`.
22    #[arg(short, long)]
23    pub output: Option<String>,
24    /// Width of the image in pixels.
25    #[arg(short, long, default_value = "1024")]
26    pub width: u32,
27    /// Height of the image in pixels.
28    #[arg(short, long, default_value = "1024")]
29    pub height: u32,
30    /// Number of samples to generate per each pixel.
31    #[arg(short, long, default_value = "64")]
32    pub samples: u32,
33    /// Maximum evaluated bounce depth for each ray.
34    #[arg(short = 'd', long, default_value = "64")]
35    pub max_depth: u32,
36    /// Rendering mode.
37    #[arg(short = 'm', long, default_value = "path-tracing")]
38    pub mode: RenderMode,
39    /// Sampler to use for rendering.
40    #[arg(long, default_value = "random")]
41    pub sampler: Sampler,
42    /// BVH construction algorithm.
43    #[arg(long, default_value = "sah")]
44    pub bvh: BvhAlgorithm,
45    /// File format selection for the output.
46    /// Multiple formats can be provided to save the same image in multiple formats.
47    #[arg(short, long, default_value = "png", num_args = 1..)]
48    pub formats: Vec<Format>,
49}

51#[derive(Copy, Clone, Debug, PartialEq, ValueEnum)]
52pub enum RenderMode {
53    /// Full path tracing, the default
54    PathTracing,
55    /// Surface normals of the first hit
56    NormalMap,
57    /// Debug view for BVH ray hit count
58    BvhTestCount,
59    /// Debug view for primitive object ray hit count
60    PrimitiveTestCount,
61}
62
63#[derive(Copy, Clone, Debug, PartialEq, ValueEnum)]
64pub enum BvhAlgorithm {
65    /// Split at the Longest Axis Midpoint of the current AABB
66    Lam,
67    /// Split based on the Surface Area Heuristic.
68    Sah,
69}

71#[derive(Copy, Clone, Debug, PartialEq, ValueEnum)]
72pub enum Format {
73    /// Portable Network Graphics, lossless, standard dynamic range
74    Png,
75    /// OpenEXR, high dynamic range

81pub(crate) fn render(
82    global_options: GlobalOptions,
83    render_options: RenderOptions,
84) -> Result<(), Box<dyn Error>> {
85    let GlobalOptions { quiet, debug } = global_options;
86    let RenderOptions {
87        ref input,
88        ref output,
89        width,
90        height,
91        samples,
92        max_depth,
93        mode,
94        sampler,
95        bvh,
96        ref formats,
97    } = render_options;
98
99    if debug {
100        tracing_subscriber::fmt()
101            .with_max_level(Level::DEBUG)
102            .with_timer(UtcTime::rfc_3339())
103            .init();
104        debug!("Debug logging enabled");
105    } else {
106        tracing_subscriber::fmt()
107            .with_max_level(Level::ERROR)
108            .with_timer(UtcTime::rfc_3339())
109            .init();
110    }
111
112    // Pretty printing output, unless in quiet mode
113    if !quiet {
114        println!("clovers üçÄ path tracing renderer");
115        println!();
116        println!("{width}x{height} resolution");
117        if mode == RenderMode::NormalMap {
118            println!("rendering a normalmap");
119        } else {
120            println!("{samples} samples per pixel");
121            println!("using the {sampler} sampler");
122            println!("{max_depth} max bounce depth");
123        }
124        println!(); // Empty line before progress bar
125    }
126
127    if sampler == Sampler::Blue && !([1, 2, 4, 8, 16, 32, 64, 128, 256].contains(&samples)) {
128        panic!("the blue sampler only supports the following sample-per-pixel counts: [1, 2, 4, 8, 16, 32, 64, 128, 256]");
129    }
130
131    // TODO: improve ergonomics?
132    let bvh_algorithm: clovers::bvh::BvhAlgorithm = match bvh {
133        BvhAlgorithm::Lam => clovers::bvh::BvhAlgorithm::Lam,
134        BvhAlgorithm::Sah => clovers::bvh::BvhAlgorithm::Sah,
135    };
136
137    info!("Reading the scene file");
138    let path = Path::new(&input);
139    let scene = match path.extension() {
140        Some(ext) => match &ext.to_str() {
141            Some("json") => initialize(path, bvh_algorithm, width, height),
142            _ => panic!("Unknown file type"),
143        },
144        None => panic!("Unknown file type"),
145    }?;
146
147    info!("Calling draw()");
148    let start = Instant::now();
149    let pixelbuffer = draw_cpu::draw(&global_options, &render_options, &scene, sampler);
150    let duration = Instant::now() - start;
151    let duration = format_duration(duration);
152    info!("Finished render in {}", duration);
153    if !quiet {
154        println!("Finished render in {}", duration);
155    }
156
157    for format in formats {
158        let extension = match format {
159            Format::Png => "png",
160            Format::Exr => "exr",
161        };
162
163        let target = match output {
164            Some(filename) => format!("{filename}.{extension}"),
165            None => {
166                // Default to using a timestamp & `renders/` directory
167                let timestamp = OffsetDateTime::now_utc().unix_timestamp();
168                fs::create_dir_all("renders")?;
169                format!("renders/{timestamp}.{extension}")
170            }
171        };
172
173        match format {
174            Format::Png => write::png(&pixelbuffer, &target, &duration, &render_options),
175            Format::Exr => write::exr(&pixelbuffer, width, height, &target),
176        }?;
177
178        info!("Image saved to {}", target);
179        println!("Image saved to: {}", target);
180    }
181
182    Ok(())
183}
