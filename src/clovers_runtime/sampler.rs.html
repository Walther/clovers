1//! Sampler architecture for the renderer, based on the sampling infrastructure described in the book Physically Based Rendering, chapter [8.3 Sampling Interface](https://pbr-book.org/4ed/Sampling_and_Reconstruction/Sampling_Interface#Sampler)
2
3use std::fmt::Display;
4
5use clap::ValueEnum;
6use clovers::{wavelength::Wavelength, Float, Vec2};
7
8pub mod blue;
9pub mod random;

11pub trait SamplerTrait<'scene> {
12    // TODO: better types

25
26/// A collection of random values to be used for each sample. Returned as a struct to ensure the correct sampling order for the underlying source of randomness.
27pub struct Randomness {
28    /// Intra-pixel `(x,y)` offset, both in range `[0..1]`. Used for antialiasing.
29    pub pixel_offset: Vec2,
30    /// The `(x,y)` offset used in the lens equations for aperture / depth-of-field simulation. The coordinates are within the range `[-0.5..0.5]` and within a unit disk.
31    pub lens_offset: Vec2,
32    /// The time of the ray, in range `[0..1]`
33    pub time: Float,
34    /// Wavelength of the ray
35    pub wavelength: Wavelength,
36}
37
38/// Enum of the supported samplers.
39#[derive(Copy, Clone, Debug, PartialEq, ValueEnum)]
40pub enum Sampler {
41    /// Blue noise based sampler
42    Blue,
43    /// Random number generator based sampler
44    Random,
45}
46
47impl Display for Sampler {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        let s = match self {
50            Sampler::Blue => "blue",
51            Sampler::Random => "random",
52        };
53        write!(f, "{s}")
54    }
55}
56
57/// Various sampling dimensions used by the samplers
58#[derive(Clone, Debug)]
59pub enum SamplerDimension {
60    PixelOffsetX,
61    PixelOffsetY,
62    LensOffsetX,
63    LensOffsetY,
64    Time,
65    Wavelength,
66}
