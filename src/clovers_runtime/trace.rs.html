1//! An opinionated colorize method. Given a [Ray] and a [Scene], evaluates the ray's path and returns a color.
2
3use clovers::{
4    hitable::HitableTrait,
5    materials::MaterialType,
6    pdf::{HitablePDF, MixturePDF, PDFTrait, PDF},
7    ray::Ray,
8    scenes::Scene,
9    spectrum::spectral_powers,
10    wavelength::{rotate_wavelength, WAVE_SAMPLE_COUNT},
11    Float, EPSILON_SHADOW_ACNE,
12};
13use nalgebra::Unit;
14use rand::rngs::SmallRng;
15
16use crate::sampler::SamplerTrait;

18/// The main path tracing function. Sends a [`Ray`] to the [`Scene`], sees if it hits anything, and eventually returns a spectral intensity. Taking into account the [Material](clovers::materials::Material) that is hit, the method recurses with various adjustments, with a new [`Ray`] started from the location that was hit.
19#[must_use]
20#[allow(clippy::only_used_in_recursion)] // TODO: use sampler in more places!
21pub fn trace(
22    ray: &Ray,
23    scene: &Scene,
24    depth: u32,
25    max_depth: u32,
26    rng: &mut SmallRng,
27    sampler: &dyn SamplerTrait,
28) -> [Float; WAVE_SAMPLE_COUNT] {
29    let hero = ray.wavelength;
30    let wavelengths = rotate_wavelength(hero);
31
32    let bg = spectral_powers(scene.background, wavelengths);
33
34    // Have we reached the maximum recursion i.e. ray bounce depth?
35    if depth > max_depth {
36        // Ray bounce limit reached, early return zero emissivity
37        return [0.0; WAVE_SAMPLE_COUNT];
38    }
39
40    // Send the ray to the scene, and see if it hits anything.
41    // distance_min is set to an epsilon to avoid "shadow acne" that can happen when set to zero
42    let Some(hit_record) = scene
43        .bvh_root
44        .hit(ray, EPSILON_SHADOW_ACNE, Float::MAX, rng)
45    else {
46        // If the ray hits nothing, early return the background color as emissivity
47        return bg;
48    };
49
50    // Get the emitted color from the surface that we just hit
51    let emitted =
52        std::array::from_fn(|i| hit_record.material.emit(ray, wavelengths[i], &hit_record));
53
54    // Do we scatter?
55    let Some(scatter_record) = hit_record.material.scatter(ray, &hit_record, rng) else {
56        // No scatter, early return the emitted color only
57        return emitted;
58    };
59    // We have scattered, and received an attenuation from the material
60    // Are we on a dispersive material? If so, terminate other wavelengths
61    let attenuations = match hit_record.material.is_wavelength_dependent() {
62        true => {
63            let mut ret = [0.0; WAVE_SAMPLE_COUNT];
64            ret[0] = hit_record.material.color(ray, hero, &hit_record);
65            ret
66        }
67        false => {
68            std::array::from_fn(|i| hit_record.material.color(ray, wavelengths[i], &hit_record))
69        }
70    };
71
72    // Check the material type and recurse accordingly:
73    match scatter_record.material_type {
74        MaterialType::Specular => {
75            // If we hit a specular material, recurse with a specular ray, and multiply it with the attenuation
76            let scatter_ray = scatter_record.specular_ray.unwrap();
77            let specular = trace(&scatter_ray, scene, depth + 1, max_depth, rng, sampler);
78            std::array::from_fn(|i| specular[i] * attenuations[i])
79        }
80        MaterialType::Diffuse => {
81            // Multiple Importance Sampling:
82
83            // Create a new PDF object from the priority hitables of the scene, given the current hit_record position
84            let light_ptr =
85                PDF::HitablePDF(HitablePDF::new(&scene.mis_bvh_root, hit_record.position));
86
87            // Create a mixture PDF from the above + the PDF from the scatter_record
88            let mixture_pdf = MixturePDF::new(light_ptr, scatter_record.pdf_ptr);
89
90            // Generate a direction for the scattering ray to go towards, weighed by the mixture PDF
91            let direction = Unit::new_normalize(mixture_pdf.generate(rng));
92
93            // Create the ray
94            let scatter_ray = Ray {
95                origin: hit_record.position,
96                direction,
97                time: ray.time,
98                wavelength: ray.wavelength,
99            };
100
101            // Get the distribution value for the PDF
102            // TODO: improve correctness & optimization!
103            let mis_pdf_value = mixture_pdf.value(direction, hero, ray.time, rng);
104
105            // Calculate the PDF weighting for the scatter
106            // TODO: improve correctness & optimization!
107            let Some(scattering_pdf) = hit_record
108                .material
109                .scattering_pdf(&hit_record, &scatter_ray)
110            else {
111                // No scatter, only emit
112                return emitted;
113            };
114
115            // Recurse for the scattering ray
116            let recurse = trace(&scatter_ray, scene, depth + 1, max_depth, rng, sampler);
117            std::array::from_fn(|i| {
118                emitted[i] + recurse[i] * attenuations[i] * scattering_pdf / mis_pdf_value
119            })
120        }
121    }
122}
