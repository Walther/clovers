20/// A serialized representation of a [Scene].
21pub struct SceneFile {
22    time_0: Float,
23    time_1: Float,
24    background_color: Srgb,
25    camera: CameraInit,
26    objects: Vec<Object>,
27    #[serde(default)]
28    materials: Vec<SharedMaterial>,
29}
30
31impl SceneFile {
32    /// Initializes a new [Scene] instance by parsing the contents of a [`SceneFile`] structure and then using those details to construct the [Scene].
33    #[must_use]
34    pub fn initialize<'scene>(
35        scene_file: SceneFile,
36        bvh_algorithm: BvhAlgorithm,
37        width: u32,
38        height: u32,
39    ) -> Scene<'scene> {
40        let time_0 = scene_file.time_0;
41        let time_1 = scene_file.time_1;
42
43        let background: Xyz = scene_file.background_color.into_color_unclamped();
44        let background: Xyz<E> = background.adapt_into();
45
46        #[allow(clippy::cast_precision_loss)]
47        let camera = Camera::new(
48            scene_file.camera.look_from,
49            scene_file.camera.look_at,
50            scene_file.camera.up,
51            scene_file.camera.vertical_fov,
52            width as Float / height as Float,
53            scene_file.camera.aperture,
54            scene_file.camera.focus_distance,
55            time_0,
56            time_1,
57        );
58        let mut materials = scene_file.materials;
59        materials.push(SharedMaterial::default());
60        let materials = Box::leak(Box::new(materials));
61
62        info!("Creating a flattened list from the objects");
63        let mut hitables: Vec<Hitable> = Vec::new();
64        let mut priority_hitables: Vec<Hitable> = Vec::new();
65
66        // TODO: this isn't the greatest ergonomics, but it gets the job done for now
67        for object in scene_file.objects {
68            let priority = match &object {
69                Object::Boxy(i) => i.priority,
70                Object::ConstantMedium(i) => i.priority,
71                Object::MovingSphere(i) => i.priority,
72                Object::ObjectList(i) => i.priority,
73                Object::Quad(i) => i.priority,
74                Object::RotateY(i) => i.priority,
75                Object::Sphere(i) => i.priority,
76                Object::STL(i) => i.priority,
77                Object::PLY(i) => i.priority,
78                Object::GLTF(i) => i.priority,
79                Object::Translate(i) => i.priority,
80                Object::Triangle(i) => i.priority,
81            };
82            let hitable = object_to_hitable(object, materials);
83            // If we get a HitableList, flatten it
84            if let Hitable::HitableList(l) = hitable {
85                for h in l.hitables {
86                    if let Hitable::HitableList(_) = h {
87                        panic!("Recursing too hard in scene initialization")
88                    }
89                    if priority {
90                        // FIXME: remove this and then remove Clone from Hitable
91                        priority_hitables.push(h.clone());
92                    }
93                    hitables.push(h);
94                }
95            } else {
96                // Otherwise, push as-is
97                if priority {
98                    // FIXME: remove this and then remove Clone from Hitable
99                    priority_hitables.push(hitable.clone());
100                }
101                hitables.push(hitable);
102            }
103        }
104        info!("All objects parsed into hitables");
105        info!("Building the BVH root for hitables");
106        let bvh_root = BVHNode::from_list(bvh_algorithm, hitables);
107        info!("Building the MIS BVH root for priority hitables");
108        let mis_bvh_root = Hitable::BVHNode(BVHNode::from_list(bvh_algorithm, priority_hitables));
109        info!("BVH root nodes built");
110
111        Scene {
112            camera,
113            bvh_root,
114            mis_bvh_root,
115            background,
116        }
117    }
118}
